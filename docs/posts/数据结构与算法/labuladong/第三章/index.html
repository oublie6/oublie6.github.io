var primes []bool

func countPrimes(n){
  primes=make([]bool,n+1)
  count:=0
  for i:=2;i<=n;i++{primes[i]=true}
  for i:=2;i<=n;i++{
    if primes[i]!=true{continue}
    count++
    tmp:=i
    for tmp<=n{
      primes[tmp]=false
      tmp+=i
    }
  }
  return count
}

var base int

func mypow(a,k int)int{
  if k==0 {return 1}
  if k==1 {return a}

  a %= base
  if k&1==1{
    tmp:=mypow(a,k-1)%base
    return tmp*a
  }else{
    tmp:=mypow(a,k/2)%base
    return tmp*tmp
  }
}
